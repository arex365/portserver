import requests
import datetime
import time
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm import tqdm

BASE_URL = "https://fapi.binance.com"
prodMode = True

def serial(data):
    if prodMode == False:
        print(data)
def get_active_futures_symbols():
    url = f"{BASE_URL}/fapi/v1/exchangeInfo"
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        data = response.json()

        return [
            s["symbol"]
            for s in data["symbols"]
            if s["contractType"] == "PERPETUAL"
            and s["status"] == "TRADING"
            and s["quoteAsset"] == "USDT"
        ]
    except Exception:
        return []


def fetch_candle_data(symbol, start_time_ms):
    url = f"{BASE_URL}/fapi/v1/klines"
    params = {
        "symbol": symbol,
        "interval": "1d",
        "startTime": start_time_ms,
        "limit": 1
    }

    try:
        response = requests.get(url, params=params, timeout=10)
        response.raise_for_status()
        data = response.json()

        if not data:
            return None

        open_price = float(data[0][1])
        close_price = float(data[0][4])

        change_percent = ((close_price - open_price) / open_price) * 100 if open_price > 0 else 0

        return {
            "symbol": symbol,
            "priceChangePercent": change_percent
        }

    except Exception:
        return None


def get_all_futures_data(symbols, start_time_ms, max_workers=1):
    results = []

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = [
            executor.submit(fetch_candle_data, symbol, start_time_ms)
            for symbol in symbols
        ]

        completed_futures = as_completed(futures)
        iterator = completed_futures if prodMode else tqdm(completed_futures, total=len(futures), desc="Fetching")

        for future in iterator:
            result = future.result()
            if result:
                results.append(result)

    return results


def get_top_gainers(futures_data, top_n=1):
    sorted_data = sorted(
        futures_data,
        key=lambda x: x["priceChangePercent"],
        reverse=True
    )
    return sorted_data[:top_n]


def get_top_losers(futures_data, top_n=1):
    sorted_data = sorted(
        futures_data,
        key=lambda x: x["priceChangePercent"],
        reverse=False
    )
    return sorted_data[:top_n]


def get_short_candidates(futures_data, threshold=15):
    return [
        coin["symbol"]
        for coin in futures_data
        if coin["priceChangePercent"] > threshold
    ]


def get_long_candidates(futures_data, threshold=-15):
    return [
        coin["symbol"]
        for coin in futures_data
        if coin["priceChangePercent"] < threshold
    ]


def run():
    serial("\nChecking threshold coins (+15% short / -15% long)...")

    symbols = get_active_futures_symbols()
    if not symbols:
        return [], []

    today = datetime.datetime.utcnow().replace(
        hour=0, minute=0, second=0, microsecond=0
    )
    start_time_ms = int(today.timestamp() * 1000)

    futures_data = get_all_futures_data(symbols, start_time_ms)

    if not futures_data:
        return [], []

    short_candidates = get_short_candidates(futures_data, threshold=15)
    long_candidates = get_long_candidates(futures_data, threshold=-15)

    return short_candidates, long_candidates


# ===== Trade Management Logic =====

short_coins = []
long_coins = []


def CloseTrade(coin):
    return True
    #remove USDT from coin
    coin = coin.replace("USDT", "")
    url = f"http://f1.itsarex.com:5007/manage/{coin}?tableName=Top"
    payload = {
        'Action': "CloseLong",
        "positionSize": 100
    }    
    try:
        response = requests.post(url, json=payload)
        serial(f"? Closing trade for {coin}: {response.text}")
        return response.status_code == 200
    except Exception as e:
        serial(f"? Failed to close trade for {coin}: {e}")
        return False


def OpenTrade(coin):
    coin = coin.replace("USDT", "")
    url = f"http://f1.itsarex.com:5007/manage/{coin}?tableName=Crash"
    url2 = f"http://f1.itsarex.com:5007/manage/{coin}?tableName=CrashRev"
    payload = {
        'Action': "Long",
        "positionSize": 100
    }
    payload2 = {
        'Action': "Short",
        "positionSize": 100
    }

    try:
        response = requests.post(url, json=payload)
        response2 = requests.post(url2,json=payload2)
        serial(f"? Opening trade for {coin}: {response.text}")
        return response.status_code == 200
    except Exception as e:
        serial(f"? Failed to open trade for {coin}: {e}")
        return False

def SetShortCoins(new_short_coins):
    global short_coins

    new_short_coins = list(dict.fromkeys(new_short_coins))

    # Close short positions that are no longer in threshold list
    for coin in short_coins:
        if coin not in new_short_coins:
            CloseShortTrade(coin)

    # Open new short positions for coins above +15%
    for coin in new_short_coins:
        if coin not in short_coins:
            OpenShortTrade(coin)

    short_coins = new_short_coins.copy()

    serial(f"Current Active Short Coins: {short_coins} (Total: {len(short_coins)})")


def SetLongCoins(new_long_coins):
    global long_coins

    new_long_coins = list(dict.fromkeys(new_long_coins))

    # Close long positions that are no longer in threshold list
    for coin in long_coins:
        if coin not in new_long_coins:
            CloseTrade(coin)

    # Open new long positions for coins below -15%
    for coin in new_long_coins:
        if coin not in long_coins:
            OpenTrade(coin)

    long_coins = new_long_coins.copy()

    serial(f"Current Active Long Coins: {long_coins} (Total: {len(long_coins)})")


if __name__ == "__main__":
    while True:
